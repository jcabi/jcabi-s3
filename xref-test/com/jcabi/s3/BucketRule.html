5   package com.jcabi.s3;
6   
7   import com.jcabi.log.Logger;
8   import com.jcabi.s3.cached.CdRegion;
9   import com.jcabi.s3.retry.ReRegion;
10  import java.util.Locale;
11  import org.apache.commons.lang3.RandomStringUtils;
12  import org.junit.jupiter.api.Assumptions;
13  import org.junit.jupiter.api.extension.AfterEachCallback;
14  import org.junit.jupiter.api.extension.BeforeEachCallback;
15  import org.junit.jupiter.api.extension.ExtensionContext;
16  import software.amazon.awssdk.services.s3.S3Client;
17  import software.amazon.awssdk.services.s3.model.CreateBucketRequest;
18  import software.amazon.awssdk.services.s3.model.DeleteBucketRequest;
19  import software.amazon.awssdk.services.s3.model.HeadBucketRequest;
20  import software.amazon.awssdk.services.s3.model.NoSuchBucketException;
21  
22  /**
23   * Extension that creates and drops an AWS subj.
24   *
25   * @since 0.3
26   */
27  final class BucketRule implements BeforeEachCallback, AfterEachCallback {
28  
29      /**
30       * AWS key.
31       */
32      private static final String KEY =
33          System.getProperty("failsafe.s3.key");
34  
35      /**
36       * AWS secret.
37       */
38      private static final String SECRET =
39          System.getProperty("failsafe.s3.secret");
40  
41      /**
42       * Bucket we're working with.
43       */
44      private transient Bucket subj;
45  
46      @Override
47      public void beforeEach(final ExtensionContext ctx) throws Exception {
48          Assumptions.assumeTrue(
49              BucketRule.KEY != null && !BucketRule.KEY.isEmpty(),
50              "system property failsafe.s3.key is not set, skipping"
51          );
52          this.create();
53      }
54  
55      @Override
56      public void afterEach(final ExtensionContext ctx) throws Exception {
57          if (this.subj != null) {
58              this.drop();
59          }
60      }
61  
62      /**
63       * Get bucket.
64       * @return Bucket
65       */
66      public Bucket bucket() {
67          return this.subj;
68      }
69  
70      /**
71       * Create S3 subj.
72       * @throws Exception If fails
73       */
74      private void create() throws Exception {
75          final Region region = new CdRegion(
76              new ReRegion(
77                  new Region.Simple(BucketRule.KEY, BucketRule.SECRET)
78              )
79          );
80          // @checkstyle MagicNumberCheck (3 line)
81          final String name = String.format(
82              "%s.s3.jcabi.com",
83              RandomStringUtils.randomAlphabetic(5)
84                  .toLowerCase(Locale.ENGLISH)
85          );
86          this.subj = region.bucket(name);
87          final S3Client aws = this.subj.region().aws();
88          aws.createBucket(
89              CreateBucketRequest.builder().bucket(name).build()
90          );
91          Logger.info(this, "S3 bucket %s created", name);
92      }
93  
94      /**
95       * Drop S3 subj.
96       * @throws Exception If fails
97       */
98      private void drop() throws Exception {
99          final S3Client aws = this.subj.region().aws();
100         try {
101             aws.headBucket(
102                 HeadBucketRequest.builder()
103                     .bucket(this.subj.name())
104                     .build()
105             );
106             aws.deleteBucket(
107                 DeleteBucketRequest.builder()
108                     .bucket(this.subj.name())
109                     .build()
110             );
111             Logger.info(this, "S3 bucket %s deleted", this.subj.name());
112         } catch (final NoSuchBucketException ex) {
113             Logger.info(
114                 this, "S3 bucket %s already gone", this.subj.name()
115             );
116         }
117     }
118 
119 }
