5   package com.jcabi.s3;
6   
7   import org.hamcrest.MatcherAssert;
8   import org.hamcrest.Matchers;
9   import org.junit.jupiter.api.Test;
10  import org.junit.jupiter.api.extension.RegisterExtension;
11  import software.amazon.awssdk.services.s3.S3Client;
12  import software.amazon.awssdk.services.s3.model.HeadBucketRequest;
13  
14  /**
15   * Integration case for {@link Region}.
16   *
17   * @since 0.1
18   */
19  final class RegionITCase {
20  
21      /**
22       * Bucket we're working with.
23       * @checkstyle VisibilityModifier (3 lines)
24       */
25      @RegisterExtension
26      final transient BucketRule rule = new BucketRule();
27  
28      /**
29       * Region can connect to AWS and check bucket existence.
30       * @throws Exception If fails
31       */
32      @Test
33      void connectsToAmazon() throws Exception {
34          final Bucket bucket = this.rule.bucket();
35          final S3Client aws = bucket.region().aws();
36          MatcherAssert.assertThat(
37              "should be true",
38              aws.headBucket(
39                  HeadBucketRequest.builder()
40                      .bucket(bucket.name())
41                      .build()
42              ),
43              Matchers.notNullValue()
44          );
45      }
46  
47  }
