5   package com.jcabi.s3;
6   
7   import com.jcabi.aspects.Loggable;
8   import com.jcabi.log.Logger;
9   import java.io.IOException;
10  import java.util.Iterator;
11  import lombok.EqualsAndHashCode;
12  import software.amazon.awssdk.services.s3.S3Client;
13  import software.amazon.awssdk.services.s3.model.DeleteObjectRequest;
14  import software.amazon.awssdk.services.s3.model.HeadBucketRequest;
15  import software.amazon.awssdk.services.s3.model.NoSuchBucketException;
16  import software.amazon.awssdk.services.s3.model.S3Exception;
17  
18  /**
19   * Amazon S3 bucket.
20   *
21   * @since 0.1
22   */
23  @EqualsAndHashCode(of = { "regn", "bkt" })
24  @Loggable(Loggable.DEBUG)
25  @SuppressWarnings("PMD.TooManyMethods")
26  final class AwsBucket implements Bucket {
27  
28      /**
29       * Region we're in.
30       */
31      private final transient Region regn;
32  
33      /**
34       * Bucket name.
35       */
36      private final transient String bkt;
37  
38      /**
39       * Public ctor.
40       * @param reg Region we're in
41       * @param name Bucket name
42       */
43      AwsBucket(final Region reg, final String name) {
44          this.regn = reg;
45          this.bkt = name;
46      }
47  
48      @Override
49      public String toString() {
50          return this.bkt;
51      }
52  
53      @Override
54      public Region region() {
55          return this.regn;
56      }
57  
58      @Override
59      public String name() {
60          return this.bkt;
61      }
62  
63      @Override
64      public Ocket ocket(final String key) {
65          return new AwsOcket(this, key);
66      }
67  
68      @Override
69      public boolean exists() throws IOException {
70          final S3Client aws = this.regn.aws();
71          boolean result = true;
72          try {
73              aws.headBucket(
74                  HeadBucketRequest.builder().bucket(this.bkt).build()
75              );
76          } catch (final NoSuchBucketException ex) {
77              result = false;
78          } catch (final S3Exception ex) {
79              throw new IOException(
80                  String.format(
81                      "failed to check existence of '%s' bucket",
82                      this.bkt
83                  ),
84                  ex
85              );
86          }
87          Logger.debug(
88              this, "Does bucket '%s' exist? %b", this.bkt, result
89          );
90          return result;
91      }
92  
93      @Override
94      @SuppressWarnings("PMD.GuardLogStatement")
95      public void remove(final String key) throws IOException {
96          try {
97              final S3Client aws = this.regn.aws();
98              final long start = System.currentTimeMillis();
99              aws.deleteObject(
100                 DeleteObjectRequest.builder()
101                     .bucket(this.bkt)
102                     .key(key)
103                     .build()
104             );
105             Logger.info(
106                 this,
107                 "ocket '%s' removed in bucket '%s' in %[ms]s",
108                 key, this.bkt, System.currentTimeMillis() - start
109             );
110         } catch (final S3Exception ex) {
111             throw new IOException(
112                 String.format(
113                     "failed to remove '%s' bucket",
114                     this.bkt
115                 ),
116                 ex
117             );
118         }
119     }
120 
121     @Override
122     public Iterable<String> list(final String pfx) {
123         return new Iterable<String>() {
124             @Override
125             public Iterator<String> iterator() {
126                 return new AwsListIterator(
127                     AwsBucket.this.regn, AwsBucket.this.bkt, pfx
128                 );
129             }
130         };
131     }
132 
133     @Override
134     public int compareTo(final Bucket bucket) {
135         return this.name().compareTo(bucket.name());
136     }
137 
138 }
